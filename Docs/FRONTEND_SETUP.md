# Frontend Setup Script
# This script contains the commands to initialize the React + TypeScript + Vite frontend

# Step 1: Create Vite project with React and TypeScript
# Run this command from the KhaoGully-Main directory:
# npm create vite@latest Frontend -- --template react-ts

# Step 2: Install dependencies
# cd Frontend
# npm install

# Step 3: Install additional required packages
# npm install react-router-dom axios
# npm install @types/react-router-dom --save-dev

# Step 4: Install Tailwind CSS (optional but recommended)
# npm install -D tailwindcss postcss autoprefixer
# npx tailwindcss init -p

# Step 5: Run development server
# npm run dev

# The frontend will be available at: http://localhost:5173
